#### Array of objects
```javascript
const courses = [
  { id: 1, name: 'course 1' },
  { id: 2, name: 'course 2' },
  { id: 3, name: 'course 3' }
];

// Read course
const course = courses.find(c => c.id === parseInt(req.params.id));

// Create course
const course = {
  id: courses.length + 1,
  name: req.body.name
};
courses.push(course);
  
// Update course
course.name = req.body.name;

// Delete course
const index = courses.indexOf(course);
courses.splice(index, 1);
```
