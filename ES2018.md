#### Array
1. Read array items
```
  const list = [1, 2, 3, 4, 5];
  list[1];                 // 2
  list.indexOf(2);         // 0
```
2. Add array items
```
  const list = [1, 2, 3, 4, 5];
  // Mutative
  list.unshift(6);         // list = [6, 1, 2, 3, 4, 5]
  list.splice(2, 0, 6);    // list = [1, 2, 6, 3, 4, 5]
  
  // Immutable
  const newList = list.concat([6, 7]);     // newList = [1, 2, 3, 4, 5, 6, 7]
```
3. Replace items
```
  const list = [1, 2, 3, 4, 5];
  // Mutative
  list.splice(2, 1, 6);    // list = [1, 2, 6, 4, 5]
```
4. Remove items
```
  const list = [1, 2, 3, 4, 5];
  // Mutative
  list.pop();              // list = [1, 2, 3, 4]
  list.shift();            // list = [2, 3, 4, 5]
  list.splice(2, 1);       // list = [1, 2, 4, 5]
```
4. Subsets
```
  const list = [1, 2, 3, 4, 5];
  // Mutative
  // array.splice(start[, deleteCount[, item1[, item2[, ...]]]])
  const sub = list.splice(1);     // sub = [2, 3, 4, 5]  list = [1]
  const sub = list.splice(1, 2);  // sub = [2, 3]        list = [1, 4, 5]
  
  // Immutable
  // arr.slice([begin[, end]])
  const sub = list.slice(0, 1);   // sub = [1]           list = [1, 2, 3, 4, 5]
  const sub = list.slice(1);      // sub = [2, 3, 4, 5]  list = [1, 2, 3, 4, 5]
  const sub = list.slice(1, 2);   // sub = [2]           list = [1, 2, 3, 4, 5]
```
